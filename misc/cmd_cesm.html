Run CESM simulation

Run CESM

usage: python scripts/run_cesm.py [-h] [-case CASE] [-aqua] [-drycore]
                                  [-drycore_topo] [-queue QUEUE] [-cam]
                                  [-stop_opt STOP_OPT] [-stop_n STOP_N] [-run]
                                  [-ntasks NTASKS] [-out_tstep OUT_TSTEP]
                                  [-walltime WALLTIME]
                                  [-start_date START_DATE]
                                  [-stop_date STOP_DATE] [-project PROJECT]
                                  [-data_dir DATA_DIR] [-sst_file SST_FILE]
                                  [-topo_file TOPO_FILE]
                                  [-tref_file TREF_FILE]
                                  [-cime_out_dir CIME_OUT_DIR]
                                  [-docn_stream_file DOCN_STREAM_FILE]
                                  [-docn_mode DOCN_MODE] [-rebuild] -config
                                  CONFIG
